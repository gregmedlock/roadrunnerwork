00086 inline unsigned long MTRand_int32::twiddle(unsigned long u, unsigned long v) const {
00087   return (((u & 0x80000000UL) | (v & 0x7FFFFFFFUL)) >> 1)
00088     ^ ((v & 1UL) ? 0x9908B0DFUL : 0x0UL);
00089 }

00091 inline unsigned long MTRand_int32::rand_int32() const { // generate 32 bit random int
00092   if (p == n) gen_state(); // new state vector needed
00093 // gen_state() is split off to be non-inline, because it is only called once
00094 // in every 624 calls and otherwise irand() would become too big to get inlined
00095   unsigned long x = state[p++];
00096   x ^= (x >> 11);
00097   x ^= (x << 7) & 0x9D2C5680UL;
00098   x ^= (x << 15) & 0xEFC60000UL;
00099   return x ^ (x >> 18);
00100 }
