        RRHandle rrInstance = getRRInstance();

        printf("loading model file %s\n", argv[1]);
 
        if (!loadSBMLFromFile(argv[1])) {
           printf ("Error while loading SBML file\n");
           printf ("Error message: %s\n", getLastError());
           exit();
        }
